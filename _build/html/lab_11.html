# turn the population columns into rows
populations = pd.melt(
    pop_by_cd,
    id_vars=["borough", "cd_number", "cd_name"],
    var_name="year",
    value_name="population",
)

populations["year"] = populations["year"].str.extract(r"(\d{4})").astype(int)

borough_pop = (
